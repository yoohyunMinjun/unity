CREATE TABLE users (
    id INT(2) NOT NULL,
    username VARCHAR(8),
    password CHAR(16),
    PRIMARY KEY(id)
);

CREATE TABLE score (
    score_id INT AUTO_INCREMENT,
    user_id INT(2),
    score INT,
    game_progress VARCHAR(20),
    meal_frequency INT DEFAULT 0,
    PRIMARY KEY(score_id),
    FOREIGN KEY(user_id) REFERENCES users(id)
);
